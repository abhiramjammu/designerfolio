<script>
  // --- STATE ---
  let isPanel1VideoHovered = false;
  let isPanel2VideoHovered = false;
  let isPanel5VideoHovered = false;
  const sliderStates = {
    'panel1-slider-box': { currentSlide: 0, isSliding: false },
    'video-slider-box':  { currentSlide: 0, isSliding: false },
    'panel5-slider-box': { currentSlide: 0, isSliding: false },
    'panel4-slider-box': { currentSlide: 0, isSliding: false }
  };

  // --- FISHEYE: play on hover, pause on leave ---
  document.querySelectorAll('.fisheye-card').forEach(card => {
    const video = card.querySelector('video');
    const btn   = card.querySelector('.sound-btn');
    let soundOn = false;

    card.addEventListener('mouseenter', () => {
      if(video) { video.muted = true; video.play().catch(()=>{}); }
    });
    card.addEventListener('mouseleave', () => {
      if(video) { video.muted = true; }
    });
    if(btn) {
      btn.addEventListener('click', e => {
        e.stopPropagation();
        soundOn = !soundOn;
        if(video) video.muted = !soundOn;
        btn.textContent = soundOn ? '🔊' : '🔇';
      });
    }
  });

  // --- AUTO-PLAY OBSERVER for Panel 4 ---
  const autoPlayObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      const video = entry.target;
      if(entry.isIntersecting && !video.dataset.hasStarted) {
        video.muted = true;
        video.currentTime = 0;
        video.dataset.hasStarted = "true";
        video.play().catch(()=>{});
      }
    });
  }, { threshold: 0.5 });

  document.querySelectorAll('.panel-4 video').forEach(v => autoPlayObserver.observe(v));

  // --- BURST LOGIC ---
  const panel1el = document.querySelector('#panel-1');
  if(panel1el) {
    panel1el.addEventListener('mouseenter', () => {
      const sliderBox = document.getElementById('panel1-slider-box');
      if(sliderBox) {
        const track = sliderBox.querySelector('.internal-slider-track');
        const idx   = sliderStates['panel1-slider-box'].currentSlide;
        const vid   = track.children[idx].querySelector('video');
        if(vid && !isPanel1VideoHovered) {
          vid.muted = true;
          vid.play().then(() => { setTimeout(() => { if(!isPanel1VideoHovered) vid.pause(); }, 150); }).catch(()=>{});
        }
      }
    });
  }
  const panel2el = document.querySelector('#panel-2');
  if(panel2el) {
    panel2el.addEventListener('mouseenter', () => {
      const sliderBox = document.getElementById('video-slider-box');
      if(sliderBox) {
        const track = sliderBox.querySelector('.internal-slider-track');
        const idx   = sliderStates['video-slider-box'].currentSlide;
        const vid   = track.children[idx].querySelector('video');
        if(vid && !isPanel2VideoHovered) {
          vid.muted = true;
          vid.play().then(() => { setTimeout(() => { if(!isPanel2VideoHovered) vid.pause(); }, 150); }).catch(()=>{});
        }
      }
    });
  }
  const panel5el = document.querySelector('#panel-5');
  if(panel5el) {
    panel5el.addEventListener('mouseenter', () => {
      const sliderBox = document.getElementById('panel5-slider-box');
      if(sliderBox) {
        const track = sliderBox.querySelector('.internal-slider-track');
        const idx   = sliderStates['panel5-slider-box'].currentSlide;
        const vid   = track.children[idx].querySelector('video');
        if(vid && !isPanel5VideoHovered) {
          vid.muted = true;
          vid.play().then(() => { setTimeout(() => { if(!isPanel5VideoHovered) vid.pause(); }, 150); }).catch(()=>{});
        }
      }
    });
  }

  const safePlay = (video) => {
    if(!video.dataset.hasStarted) {
      video.currentTime = 0;
      video.dataset.hasStarted = "true";
    }
    if(video.ended) video.currentTime = 0;
    video.play().catch(()=>{});
  };

  // --- VIDEO BOX LOGIC ---
  document.querySelectorAll('.video-box').forEach(box => {
    if(box.id === 'panel4-slider-box') return;
    const isInternalSlider = box.querySelector('.internal-slider-track');
    const getActiveVideo = () => {
      if(isInternalSlider) {
        const boxId = box.id;
        const state = sliderStates[boxId];
        const track = box.querySelector('.internal-slider-track');
        return track.children[state ? state.currentSlide : 0].querySelector('video');
      }
      return box.querySelector('video');
    };
    const video = getActiveVideo();
    const button = box.querySelector('.sound-btn');
    const isPanel4 = box.closest('.panel-4') !== null;
    const dynamicWrapper = box.closest('.dynamic-media-wrapper');
    const isPanel1 = box.closest('#panel-1') !== null;
    const isPanel2 = box.closest('#panel-2') !== null;
    const isPanel5 = box.closest('#panel-5') !== null;

    if(video) video.muted = true;
    if(button) button.textContent = "🔇";
    let soundOn = false;

    const closeCard = () => {
      if(dynamicWrapper && dynamicWrapper.classList.contains('viewed')) {
        dynamicWrapper.classList.remove('viewed');
        soundOn = false;
        if(button) button.textContent = "🔇";
        const v = getActiveVideo();
        v.muted = true;
        safePlay(v);
      }
    };

    if(button) {
      button.addEventListener('click', e => {
        e.stopPropagation();
        const v = getActiveVideo();
        soundOn = !soundOn;
        v.muted = !soundOn;
        button.textContent = soundOn ? "🔊" : "🔇";
      });
    }

    box.addEventListener('mouseenter', () => {
      const v = getActiveVideo();
      if(isPanel1) isPanel1VideoHovered = true;
      if(isPanel2) isPanel2VideoHovered = true;
      if(isPanel5) isPanel5VideoHovered = true;
      safePlay(v);
      if(soundOn) v.muted = false;
    });

    box.addEventListener('mouseleave', () => {
      const v = getActiveVideo();
      if(isPanel1) isPanel1VideoHovered = false;
      if(isPanel2) isPanel2VideoHovered = false;
      if(isPanel5) isPanel5VideoHovered = false;
      v.muted = true;
      if(dynamicWrapper) {
        if(!soundOn) v.pause();
      } else if(isPanel4) { /* keep looping */ }
      else { if(!soundOn) v.pause(); }
    });

    let clickTimer = null;
    box.addEventListener('click', e => {
      const v = getActiveVideo();
      if(dynamicWrapper) {}
      if(e.detail === 1) {
        clickTimer = setTimeout(() => {
          soundOn = !soundOn;
          v.muted = !soundOn;
          if(button) button.textContent = soundOn ? "🔊" : "🔇";
        }, 250);
      } else if(e.detail === 2) {
        clearTimeout(clickTimer);
        if(dynamicWrapper) {
          if(dynamicWrapper.classList.contains('viewed')) closeCard();
          else { dynamicWrapper.classList.add('viewed'); safePlay(v); }
        }
      }
    });

    box.addEventListener('force-compress', () => {
      if(dynamicWrapper) {
        dynamicWrapper.classList.add('viewed');
        soundOn = false;
        if(button) button.textContent = "🔇";
      }
    });
  });

  // --- SLIDER LOGIC ---
  const sliderBoxes = [
    document.getElementById('panel1-slider-box'),
    document.getElementById('video-slider-box'),
    document.getElementById('panel5-slider-box'),
    document.getElementById('panel4-slider-box')
  ];
  sliderBoxes.forEach(sliderBox => {
    if(!sliderBox) return;
    const boxId = sliderBox.id;
    const internalTrack = sliderBox.querySelector('.internal-slider-track');
    const dots = sliderBox.closest('.slider-viewport').querySelectorAll('.dot');

    if(internalTrack && boxId !== 'panel4-slider-box') {
      const sliderVideos = internalTrack.querySelectorAll('video');
      const totalSlides = internalTrack.children.length;
      sliderVideos.forEach((v, index) => {
        v.addEventListener('ended', () => {
          v.pause();
          if(index === totalSlides - 1) {
            sliderBox.dispatchEvent(new CustomEvent('force-compress'));
            setTimeout(() => { updateSlider(boxId, 0); }, 500);
          } else {
            setTimeout(() => { updateSlider(boxId, index + 1); }, 300);
          }
        });
      });
    }

    dots.forEach(dot => {
      dot.addEventListener('click', () => {
        updateSlider(boxId, parseInt(dot.getAttribute('data-index')));
      });
    });

    let touchStartX = 0, touchEndX = 0;
    const handleSwipe = () => {
      const current = sliderStates[boxId].currentSlide;
      const totalSlides = internalTrack.children.length;
      if(touchEndX < touchStartX - 30) { if(current < totalSlides - 1) updateSlider(boxId, current + 1); }
      if(touchEndX > touchStartX + 30) { if(current > 0) updateSlider(boxId, current - 1); }
    };
    sliderBox.addEventListener('touchstart', e => { touchStartX = e.changedTouches[0].screenX; });
    sliderBox.addEventListener('touchend',   e => { touchEndX = e.changedTouches[0].screenX; handleSwipe(); });
    sliderBox.addEventListener('mousedown',  e => { touchStartX = e.screenX; });
    sliderBox.addEventListener('mouseup',    e => { touchEndX = e.screenX; handleSwipe(); });
    sliderBox.addEventListener('wheel', e => {
      if(Math.abs(e.deltaX) > 10) {
        e.preventDefault();
        const state = sliderStates[boxId];
        const totalSlides = internalTrack.children.length;
        if(!state.isSliding) {
          if(e.deltaX > 0) { if(state.currentSlide < totalSlides - 1) updateSlider(boxId, state.currentSlide + 1); }
          else             { if(state.currentSlide > 0) updateSlider(boxId, state.currentSlide - 1); }
        }
      }
    }, { passive: false });
  });

  function updateSlider(boxId, index) {
    const state = sliderStates[boxId];
    if(state.isSliding) return;
    state.isSliding = true;

    const sliderBox = document.getElementById(boxId);
    const internalTrack = sliderBox.querySelector('.internal-slider-track');
    const dots = sliderBox.closest('.slider-viewport').querySelectorAll('.dot');

    let textWrapper;
    if(boxId === 'panel4-slider-box') {
      textWrapper = document.getElementById('panel4-text-wrapper');
    } else {
      textWrapper = sliderBox.closest('.dynamic-media-wrapper').querySelector('.text-static-width');
    }
    const textGroups = textWrapper ? textWrapper.querySelectorAll('.text-group') : [];

    internalTrack.querySelectorAll('video').forEach(v => { v.pause(); v.muted = true; });
    state.currentSlide = index;
    internalTrack.style.transform = `translateX(-${state.currentSlide * 100}%)`;
    dots.forEach(d => d.classList.remove('active'));
    if(dots[state.currentSlide]) dots[state.currentSlide].classList.add('active');
    textGroups.forEach(t => t.classList.remove('active'));

    setTimeout(() => {
      if(textWrapper) {
        const activeText = textWrapper.querySelector(`.text-group[data-id="${state.currentSlide}"]`);
        if(activeText) activeText.classList.add('active');
      }
      state.isSliding = false;
      if(sliderBox.matches(':hover') || boxId === 'panel4-slider-box') {
        const activeSlide = internalTrack.children[index];
        const videosToPlay = activeSlide.querySelectorAll('video');
        videosToPlay.forEach(newVideo => {
          if(newVideo.ended) newVideo.currentTime = 0;
          newVideo.play().catch(()=>{});
          newVideo.muted = true;
        });
      }
    }, 300);
  }
</script>
